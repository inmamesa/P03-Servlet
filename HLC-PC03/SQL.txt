

CREATE TABLE SERVICIOS (
   
NUMEROSERVICIO INTEGER NOT NULL, 
NOMBRE VARCHAR(25), 
PRECIO INTEGER, 
IDT INTEGER,  
IMAGEN VARCHAR(50),
PRIMARY KEY (NUMEROSERVICIO),
    
FOREIGN KEY(IDT)REFERENCES TRABAJADORES(ID)

);
